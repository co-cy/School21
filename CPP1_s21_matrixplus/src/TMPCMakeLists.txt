cmake_minimum_required(VERSION 3.22)
project(CPP1_s21_matrixplus)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_STATIC_LIBRARY_PREFIX "")

if(CMAKE_BUILD_TYPE EQUAL "Debug")
add_compile_options(-fsanitize=address -g)
add_link_options(-fsanitize=address -g)
endif()

include_directories(.)

set(CPP1_s21_matrixplus_SOURCE
s21_matrix_oop.h core/constructor.cpp core/destructor.cpp arithmetic/merge.cpp arithmetic/mult.cpp comparison/eq.cpp other/transpose.cpp other/crop.cpp core/resize.cpp core/size.cpp operators/operators.cpp other/determiant.cpp other/complements.cpp other/inverse.cpp)

add_library(s21_matrix_oop STATIC ${CPP1_s21_matrixplus_SOURCE})
target_compile_options(s21_matrix_oop PRIVATE -Wall -Werror -Wextra -Wpedantic)

find_package(GTest REQUIRED)
include(GoogleTest)
enable_testing()

file(GLOB SOURCES test/*.cpp)
add_executable(tests ${SOURCES})
target_compile_options(tests PRIVATE -Wall -Werror -Wextra -Wpedantic)

target_link_libraries(tests GTest::gtest_main s21_matrix_oop)

gtest_discover_tests(tests)
